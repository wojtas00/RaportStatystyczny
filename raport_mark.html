## odchylenie standardowe procentowego wzrostu/spadku
odchylenie = function(rok_a, rok_b) {
  zmiana = round(100 - (rok_a / rok_b * 100), 2)
  okragl = sd(zmiana)
  odchylenie = round(okragl, 2)
  return(odchylenie)
}
